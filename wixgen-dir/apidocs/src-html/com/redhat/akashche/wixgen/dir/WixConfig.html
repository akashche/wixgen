016package com.redhat.akashche.wixgen.dir;
017
018import java.util.ArrayList;
019import java.util.UUID;
020
021import static org.apache.commons.lang.StringUtils.defaultString;
022import static org.apache.commons.lang.StringUtils.isNotEmpty;
023
024/**
025 * Configuration object for the {@link DirectoryGenerator}.
026 * Should be created from JSON file, doesn't support creation through API
027 *
028 * User: akashche
029 */
030public class WixConfig {
031    private String appName = "SPECIFY_ME";
032    private int versionMajor = 0;
033    private int versionMinor = 0;
034    private int versionMicro = 0;
035    private int versionPatch = 0;
036    private String vendor = "SPECIFY_ME";
037    private String vendorDirName = "";
038    private String installDirName = "";
039    private String helpLink = "";
040    private String productUuid = "SPECIFY_ME";
041    private boolean packageInstallPerUser = false;
042
043    private boolean useFeatureTree = false;
044    private String featureId = "_" + UUID.randomUUID().toString().replace('-', '_');
045    private boolean featureOptional = false;
046    private int featureLevel = 1;
047    private String featureDirectoryId = "INSTALLDIR";
048    private boolean featureConfigurableDirectory = true;
049    private String featureTitle = "SPECIFY_ME";
050    private String featureDescription = "SPECIFY_ME";
051
052    private String updateUuid = "SPECIFY_ME";
053    private boolean updateEnabled = true;
054    private boolean updateAllowDowngrades = false;
055    private boolean updateAllowSameVersionUpgrades = true;
056    private String updateDowngradeErrorMessage = "A later version of [ProductName] is already installed. Setup will now exit.";
057    private boolean updateIgnoreRemoveFailure = false;
058
059    private boolean win64 = true;
060    private int language = 1033;
061    private int codepage = 1252;
062    private String licenseFilePath = "SPECIFY_ME";
063    private String iconPath = "SPECIFY_ME";
064    // 493x58
065    private String topBannerBmpPath = "SPECIFY_ME";
066    // 493x312
067    private String greetingsBannerBmpPath = "SPECIFY_ME";
068
069    private ArrayList<RegistryKey> registryKeys = new ArrayList<RegistryKey>();
070    private ArrayList<EnvironmentVariable> environmentVariables = new ArrayList<EnvironmentVariable>();
071
072    public String getIconPath() {
073        return defaultString(iconPath);
074    }
075
076    public String getAppName() {
077        return defaultString(appName);
078    }
079
080    public String getVendor() {
081        return defaultString(vendor);
082    }
083
084    public String getVendorDirName() {
085        return isNotEmpty(vendorDirName) ? vendorDirName : vendor;
086    }
087
088    public String getInstallDirName() {
089        return isNotEmpty(installDirName) ? installDirName : appName;
090    }
091
092    public String getHelpLink() {
093        return defaultString(helpLink);
094    }
095
096    public String getProductUuid() {
097        return defaultString(productUuid);
098    }
099
100    public boolean isPackageInstallPerUser() {
101        return packageInstallPerUser;
102    }
103
104    public boolean isUseFeatureTree() {
105        return useFeatureTree;
106    }
107
108    public String getFeatureId() {
109        return defaultString(featureId);
110    }
111
112    public boolean isFeatureOptional() {
113        return featureOptional;
114    }
115
116    public int getFeatureLevel() {
117        return featureLevel;
118    }
119
120    public String getFeatureDirectoryId() {
121        return defaultString(featureDirectoryId);
122    }
123
124    public boolean isFeatureConfigurableDirectory() {
125        return featureConfigurableDirectory;
126    }
127
128    public String getFeatureTitle() {
129        return defaultString(featureTitle);
130    }
131
132    public String getFeatureDescription() {
133        return defaultString(featureDescription);
134    }
135
136    public String getUpgradeUuid() {
137        return defaultString(updateUuid);
138    }
139
140    public boolean isUpdateEnabled() {
141        return updateEnabled;
142    }
143
144    public boolean isUpdateAllowDowngrades() {
145        return updateAllowDowngrades;
146    }
147
148    public boolean isUpdateAllowSameVersionUpgrades() {
149        return updateAllowSameVersionUpgrades;
150    }
151
152    public String getUpdateDowngradeErrorMessage() {
153        return updateDowngradeErrorMessage;
154    }
155
156    public boolean isUpdateIgnoreRemoveFailure() {
157        return updateIgnoreRemoveFailure;
158    }
159
160    public String getVersion() {
161        return new StringBuilder()
162                .append(versionMajor)
163                .append(".")
164                .append(versionMinor)
165                .append(".")
166                .append(versionMicro)
167                .append(".")
168                .append(versionPatch)
169                .toString();
170    }
171
172    public boolean isWin64() {
173        return win64;
174    }
175
176    public String getLanguage() {
177        return Integer.toString(language);
178    }
179
180    public String getCodepage() {
181        return Integer.toString(codepage);
182    }
183
184    public String getLicenseFilePath() {
185        return defaultString(licenseFilePath);
186    }
187
188    public String getTopBannerBmpPath() {
189        return defaultString(topBannerBmpPath);
190    }
191
192    public String getGreetingsBannerBmpPath() {
193        return defaultString(greetingsBannerBmpPath);
194    }
195
196    public ArrayList<RegistryKey> getRegistryKeys() {
197        return null != registryKeys ? registryKeys : new ArrayList<RegistryKey>();
198    }
199
200    public ArrayList<EnvironmentVariable> getEnvironmentVariables() {
201        return null != environmentVariables ? environmentVariables : new ArrayList<EnvironmentVariable>();
202    }
203
204    static class RegistryKey {
205        String root;
206        String key;
207        ArrayList<RegistryValue> values = new ArrayList<RegistryValue>();
208
209        public String getRoot() {
210            return defaultString(root);
211        }
212
213        public String getKey() {
214            return defaultString(key);
215        }
216    }
217
218    static class RegistryValue {
219        String type = "string";
220        String name = "";
221        String value = "SPECIFY_ME";
222
223        public String getType() {
224            return defaultString(type);
225        }
226
227        public String getName() {
228            return defaultString(name);
229        }
230
231        public String getValue() {
232            return defaultString(value);
233        }
234    }
235
236    static class EnvironmentVariable {
237        private String name = "SPECIFY_ME";
238        private String action = "create";
239        private boolean system = true;
240        private String part = "first";
241        private String value = "SPECIFY_ME";
242
243        public String getName() {
244            return defaultString(name);
245        }
246
247        public String getAction() {
248            return defaultString(action);
249        }
250
251        public String getSystem() {
252            return system ? "yes" : "no";
253        }
254
255        public String getPart() {
256            return defaultString(part);
257        }
258
259        public String getValue() {
260            return defaultString(value);
261        }
262    }
263}
