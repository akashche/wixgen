016package com.redhat.akashche.wixgen.dir;
017
018import com.redhat.akashche.wixgen.jaxb.*;
019import com.redhat.akashche.wixgen.jaxb.Package;
020
021import java.io.File;
022import java.io.IOException;
023import java.math.BigInteger;
024import java.util.*;
025
026/**
027 * Allows to generate Wix JAXB objects from input directories
028 *
029 * User: akashche
030 */
031public class DirectoryGenerator {
032
033    /**
034     * Generates WiX installer descriptor as a JAXB object
035     *
036     * @param dir input directory to be used as installation contents
037     * @param conf configuration object
038     * @return  WiX installer descriptor as a JAXB object
039     * @throws IOException on input dir, config file or output dir access error
040     */
041    @SuppressWarnings("unchecked") // varargs fluent setter
042    public Wix createFromDir(File dir, WixConfig conf) throws IOException {
043        List<ComponentRef> comprefs = new ArrayList<ComponentRef>();
044        Collection<Object> files = createContents(dir, comprefs);
045        Collection<Object> icon = createIcon(conf);
046        Collection<Object> regs = createRegs(conf, comprefs);
047        Collection<Object> envs = createEnvs(conf, comprefs);
048        return new Wix()
049                .withProduct(new Product()
050                        .withName(conf.getAppName())
051                        .withManufacturer(conf.getVendor())
052                        .withId(conf.getProductUuid())
053                        .withUpgradeCode(conf.getUpgradeUuid())
054                        .withLanguage(conf.getLanguage())
055                        .withCodepage(conf.getCodepage())
056                        .withVersion(conf.getVersion())
057                        .withPackage(new Package()
058                                .withInstallerVersion(new BigInteger("200"))
059                                .withLanguages(conf.getLanguage())
060                                .withCompressed("yes")
061                                .withSummaryCodepage(conf.getCodepage())
062                                .withPlatform("x64"))
063                        .withAppIdOrBinaryOrComplianceCheck(new Media()
064                                .withId("1")
065                                .withCabinet("Application.cab")
066                                .withEmbedCab("yes")
067                        )
068                        .withAppIdOrBinaryOrComplianceCheck(new Directory()
069                                .withId("TARGETDIR")
070                                .withName("SourceDir")
071                                .withComponentOrDirectoryOrMerge(new Directory()
072                                        .withId("ProgramFiles64Folder")
073                                        .withComponentOrDirectoryOrMerge(new Directory()
074                                                .withId(genId())
075                                                .withName(conf.getVendorDirName())
076                                                .withComponentOrDirectoryOrMerge(new Directory()
077                                                        .withId("INSTALLDIR")
078                                                        .withName(conf.getInstallDirName())
079                                                        .withComponentOrDirectoryOrMerge((Collection) files)
080                                                        .withComponentOrDirectoryOrMerge((Collection) regs)
081                                                        .withComponentOrDirectoryOrMerge((Collection) envs)))))
082                        .withAppIdOrBinaryOrComplianceCheck(new Feature()
083                                .withId(genId())
084                                .withConfigurableDirectory("INSTALLDIR")
085                                .withComponentOrComponentGroupRefOrComponentRef((Collection) comprefs))
086                        .withAppIdOrBinaryOrComplianceCheck(new Property()
087                                .withId("WIXUI_INSTALLDIR")
088                                .withValue("INSTALLDIR"))
089                        .withAppIdOrBinaryOrComplianceCheck(new UIRef()
090                                .withId("WixUI_InstallDir"))
091                        .withAppIdOrBinaryOrComplianceCheck(new UIRef()
092                                .withId("WixUI_ErrorProgressText"))
093                        .withAppIdOrBinaryOrComplianceCheck((Collection) icon)
094                        .withAppIdOrBinaryOrComplianceCheck(new WixVariable()
095                                .withId("WixUILicenseRtf")
096                                .withValue(conf.getLicenseFilePath()))
097                        .withAppIdOrBinaryOrComplianceCheck(new WixVariable()
098                                .withId("WixUIBannerBmp")
099                                .withValue(conf.getTopBannerBmpPath()))
100                        .withAppIdOrBinaryOrComplianceCheck(new WixVariable()
101                                .withId("WixUIDialogBmp")
102                                .withValue(conf.getGreetingsBannerBmpPath()))
103                );
104    }
105
106    private String genId() {
107        return "_" + UUID.randomUUID().toString().replace('-', '_');
108    }
109
110    private List<Object> createContents(File dir, List<ComponentRef> comprefs) throws IOException {
111        List<Object> contents = new ArrayList<Object>();
112        if (!dir.isDirectory()) throw new IOException("Invalid directory: [" + dir.getAbsolutePath() + "]");
113        File[] listing = dir.listFiles();
114        if (null == listing) throw new IOException("Cannot list directory: [" + dir.getAbsolutePath() + "]");
115        for (File fi : listing) {
116            if (fi.isDirectory()) {
117                contents.add(createDirRecursive(fi, comprefs));
118            } else {
119                contents.add(createComp(fi, comprefs));
120            }
121        }
122        return contents;
123    }
124
125    private Directory createDirRecursive(File dir, List<ComponentRef> comprefs) throws IOException {
126        if (!dir.isDirectory()) throw new IOException("Invalid directory: [" + dir.getAbsolutePath() + "]");
127        Directory res = new Directory()
128                .withId(genId())
129                .withName(dir.getName());
130        File[] listing = dir.listFiles();
131        if (null == listing) throw new IOException("Cannot list directory: [" + dir.getAbsolutePath() + "]");
132        for (File fi : listing) {
133            if (fi.isDirectory()) {
134                res.withComponentOrDirectoryOrMerge(createDirRecursive(fi, comprefs));
135            } else {
136                res.withComponentOrDirectoryOrMerge(createComp(fi, comprefs));
137            }
138        }
139        return res;
140    }
141
142    private Component createComp(File file, List<ComponentRef> comprefs) throws IOException {
143        if (!file.isFile()) throw new IOException("Invalid file: [" + file.getAbsolutePath() + "]");
144        String id = genId();
145        comprefs.add(new ComponentRef()
146                .withId(id));
147        return new Component()
148                .withId(id)
149                .withWin64("yes")
150                .withGuid(UUID.randomUUID().toString())
151                .withAppIdOrCategoryOrClazz(new com.redhat.akashche.wixgen.jaxb.File()
152                        .withId(genId())
153                        .withName(file.getName())
154                        .withDiskId("1")
155                        .withSource(file.getPath())
156                        .withKeyPath("yes"));
157    }
158
159    private Collection<Object> createIcon(WixConfig conf) {
160        String id = genId();
161        return Arrays.asList(
162                new Icon()
163                        .withId(id)
164                        .withSourceFile(conf.getIconPath()),
165                new Property()
166                        .withId("ARPPRODUCTICON")
167                        .withValue(id)
168        );
169    }
170
171    @SuppressWarnings("unchecked") // varargs fluent setter
172    private Collection<Object> createRegs(WixConfig conf, List<ComponentRef> comprefs) {
173        List<Object> res = new ArrayList<Object>();
174        for (WixConfig.RegistryKey rk : conf.getRegistryKeys()) {
175            String id = genId();
176            comprefs.add(new ComponentRef()
177                    .withId(id));
178            Collection<Object> values = new ArrayList<Object>();
179            for (WixConfig.RegistryValue rv : rk.values) {
180                values.add(new RegistryValue()
181                        .withType(rv.getType())
182                        .withName(rv.getName())
183                        .withValue(rv.getValue()));
184            }
185            res.add(new Component()
186                    .withId(id)
187                    .withWin64("yes")
188                    .withGuid(UUID.randomUUID().toString())
189                    .withAppIdOrCategoryOrClazz(new RegistryKey()
190                            .withId(genId())
191                            .withForceCreateOnInstall("yes")
192                            .withRoot(RegistryRootType.fromValue(rk.getRoot()))
193                            .withKey(rk.getKey())
194                            .withRegistryKeyOrRegistryValueOrPermission((Collection) values))
195            );
196        }
197        return res;
198    }
199
200    private List<Object> createEnvs(WixConfig conf, List<ComponentRef> comprefs) {
201        List<Object> res = new ArrayList<Object>();
202        for (WixConfig.EnvironmentVariable ev : conf.getEnvironmentVariables()) {
203            String id = genId();
204            comprefs.add(new ComponentRef()
205                    .withId(id));
206            res.add(new Component()
207                    .withId(id)
208                    .withWin64("yes")
209                    .withGuid(UUID.randomUUID().toString())
210                    .withKeyPath("yes")
211                    .withAppIdOrCategoryOrClazz(new Environment()
212                            .withId(genId())
213                            .withName(ev.getName())
214                            .withAction(ev.getAction())
215                            .withSystem(ev.getSystem())
216                            .withPart("set".equals(ev.getAction()) ? ev.getPart() : null)
217                            .withValue(ev.getValue())));
218        }
219        return res;
220    }
